data("instacart") 
data_insta = instacart |>
  drop_na() |>
  filter(
    department == "personal care" & aisle != "oral hygiene" & aisle != "soap" & aisle != "vitamins supplements" 
  )

Column

Chart A

data_insta |>
  mutate(aisle = fct_reorder(aisle, order_number)) |> 
  plot_ly(
    x = ~aisle, y = ~order_number, color = ~aisle, 
    type = "box", colors = "viridis"
  )

Column

Chart B

data_insta |> 
  group_by(order_hour_of_day) |>
  mutate(mean_order_number = mean(order_number)) |>
  plot_ly(
    x = ~order_hour_of_day, y = ~mean_order_number, color = ~order_hour_of_day, 
    type = "scatter", mode = "markers", alpha = 0.5
  )

Chart C

data_insta |> 
  count(aisle) |> 
  mutate(aisle = fct_reorder(aisle, n)) |> 
  plot_ly(
    x = ~aisle, y = ~n, 
    type = "bar")
